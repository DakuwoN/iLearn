def reduceSum(array):
    if len(array) <= 1:
        return array[0]
    if len(array) == 0:
        return 0

    tempArray = []

    for i in range(len(array) - 1):
        tempArray.append(array[i] + array[i + 1])
        print()

    return reduceSum(tempArray)
