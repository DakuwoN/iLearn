def reverseString(sentence):
    reversedSolution = ""
    stack = []

    for character in sentence:

        if character == " ":
            while stack:
                reversedSolution += stack.pop()
            reversedSolution += ' '
        else:
            stack.append(character)

    while stack:
        reversedSolution += stack.pop()


    return reversedSolution
