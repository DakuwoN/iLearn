def rangedString(arr):
    result = []
    if arr == 0:
        return result

    start = end = prev = arr[0]

    for i in (range(1, len((arr)))):
        if arr[i] == prev + 1:
            end = arr[i]
        else:
            if start == end:
                result.append(str(start))
            else:
                result.append(f"{start}-{end}")
            start = end = arr[i]
        prev = arr[i]

    if start == end:
        result.append(start)
    else:
        result.append(f"{start}-{end}")
    return ','.join(result)


print(rangedString([1, 3, 4, 5, 7, 9, 10]))
